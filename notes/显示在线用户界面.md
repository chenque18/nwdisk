# 查看在线用户
客户端发送请求----服务器从数据库查询在线用户----服务器将在线用户列表发送给客户端----客户端显示在线用户列表  
只需用户名

```cpp
QStringList OpeDB::handleAllOnline()
{
    QString data=QString("select name from usrinfo where online=1");
    QSqlQuery query;
    query.exec(data);

    QStringList result;
    result.clear();
    while(query.next()){//查询结果有多行，循环读取
        result.append(query.value(0).toString());
    }
    return result;
}
```

```cpp
int OpeDB::handleSearchUsr(const char *name)
{
    if(name==nullptr)
    {
        return -1;
    }
    QString data=QString("select online from usrinfo where name='%1'").arg(name);
    QSqlQuery query;
    query.exec(data);
    if(query.next()){
        int ret=query.value(0).toInt();
        if(ret==1){
            return 1;//在线
        }
        else{
            return 0;//不在线
        }
    }
    else{
        return -1;
    }
}
```
在mytcpsocket.cpp
```cpp
case ENUM_MSG_TYPE_SEARCH_USR_REQUEST:{//服务器端处理搜索用户请求
        int ret=OpeDB::getInstance().handleSearchUsr(pdu->caData);
        PDU*repdu=makePDU(0);
        repdu->uiMsgType=ENUM_MSG_TYPE_SEARCH_USR_RESPOND;
        if(ret==-1){//3种情况：用户不存在、离线、在线
            strcpy(repdu->caData,SEARCH_USR_NO);
        }
        else if(ret==1){
            strcpy(repdu->caData,SEARCH_USR_ONLINE);
        }
        else if(ret==0){
            strcpy(repdu->caData,SEARCH_USR_OFFLINE);
        }
        write((char*)repdu,repdu->uiPDULen);
        free(repdu);
        repdu=nullptr;
        break;
    }
//协议中定义宏
#define SEARCH_USR_NO "no such people"
#define SEARCH_USR_ONLINE "online"
#define SEARCH_USR_OFFLINE "offline"
```
