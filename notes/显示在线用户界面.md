# 查看在线用户
客户端发送请求----服务器从数据库查询在线用户----服务器将在线用户列表发送给客户端----客户端显示在线用户列表  
只需用户名

由于显示用户界面需要使用socket向服务器请求，但socket产生在登陆界面，所以要将登陆界面改为单例模式，使得其他界面可以访问socket对象
tcpclient.h/cpp
```cpp
public:
    static TcpClient& getInstance();

TcpClient &TcpClient::getInstance()
{
    static TcpClient instance;
    return instance;
}
```
main.cpp使用单例模式调用规则
```cpp
    // TcpClient w;
    // w.show();
    TcpClient::getInstance().show();
```
在很多操作都要通过socket与服务器通信，所以将socket封装成函数，谁需要就调用  
tcpclient.h/cpp
```cpp
public:
    QTcpSocket &getTcpSocket();

QTcpSocket &TcpClient::getTcpSocket()
{
    return m_tcpSocket;
}
```
封装协议
现在protocol.h中定义了协议类型(服务器也要)

```cpp
    ENUM_MSG_TYPE_SEARCH_USR_REQUEST,//搜索用户
    ENUM_MSG_TYPE_SEARCH_USR_RESPOND,
```
friend.cpp
```cpp
#include"protocol.h"
#include"tcpclient.h"
void Friend::showOnline()
{
    if(m_pOnline->isHidden()){
        m_pOnline->show();
        //产生socket
        PDU*pdu=makePDU(0);
        pdu->uiMsgType=ENUM_MSG_TYPE_ALL_ONLINE_REQUEST;
        TcpClient::getInstance().getTcpSocket().write((char*)pdu,pdu->uiPDULen);
        free(pdu);
        pdu=nullptr;
    }
    else{
        m_pOnline->hide();
    }
}
```

```cpp
QStringList OpeDB::handleAllOnline()
{
    QString data=QString("select name from usrinfo where online=1");
    QSqlQuery query;
    query.exec(data);

    QStringList result;
    result.clear();
    while(query.next()){//查询结果有多行，循环读取
        result.append(query.value(0).toString());
    }
    return result;
}
```
在新的case中添加
```cpp
case ENUM_MSG_TYPE_ALL_ONLINE_REQUEST:{
        
        QStringList ret=OpeDB::getInstance().handleAllOnline();
        uint uiMsgLen=ret.size()*32;//每个用户名占32字节
        PDU *repdu=makePDU(uiMsgLen);
        repdu->uiMsgType=ENUM_MSG_TYPE_ALL_ONLINE_RESPOND;
        for(int i=0;i<ret.size();++i){
            memcpy((char*)(repdu->caMsg)+i*32,ret[i].toStdString().c_str(),ret[i].size());
        }
        write((char*)repdu,repdu->uiPDULen);//发送给客户端
        free(repdu);
        repdu=nullptr;
        break;
    }
```

void Online::showUsr(PDU *pdu)
{
    if(pdu==nullptr){
        return;
    }
    uint uiSize=pdu->uiMsgLen/32;//用户个数
    char caTmp[32];
    // 补充，如果没有 clear，你每次刷新就会把旧的数据再叠加进来
    ui->online_lw->clear(); 
    for(uint i=0;i<uiSize;i++){
        memcpy(caTmp,(char*)(pdu->caMsg)+i*32,32);
        //ui更新
        ui->online_lw->addItem(caTmp);
    }
}
