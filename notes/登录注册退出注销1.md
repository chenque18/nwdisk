```cpp
//在构造函数中连接数据库
OpeDB::OpeDB(QObject *parent)
    : QObject{parent}
{
    m_db=QSqlDatabase::addDatabase("QMYSQL");
}

void OpeDB::init()
{
    //连接数据库,本机ip地址，数据库名，用户名，密码
    m_db.setHostName("172.20.10.2");
    m_db.setDatabaseName("snetwork_disk");

    m_db.setUserName("root");         // 填入您的 MySQL 用户名，通常是 root
    m_db.setPassword("123456");  // 填入您的 MySQL 密码

    if(m_db.open()){
        //打开数据库直接复制
        QSqlQuery query;
        if(!query.exec("select * from usrinfo")) {
            qDebug() << "查询失败：" << query.lastError().text();
        }
        while(query.next())
        {
            QString data = QString("%1, %2, %3, %4")
                    .arg(query.value(0).toString())
                    .arg(query.value(1).toString())
                    .arg(query.value(2).toString())
                    .arg(query.value(3).toString());
            qDebug() << data;
            qDebug()<<"连接成功";
        }
    }
    else{
        qDebug() << "数据库连接失败，原因: " << m_db.lastError().text();
        QMessageBox::critical(nullptr,"打开数据库","打开失败");
    }
}
```
